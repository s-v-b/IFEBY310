squaresDF = spark.createDataFrame(sc.parallelize(range(1, 6))
                                  .map(lambda i: Row(single=i, double=i ** 2)))
squaresDF.write.parquet("data/test_table/key=1", 
mode="overwrite")

# Create another DataFrame in a new partition directory,
# adding a new column and dropping an existing column
cubesDF = spark.createDataFrame(sc.parallelize(range(6, 11))
                                .map(lambda i: Row(single=i, triple=i ** 3)))

cubesDF.write.parquet("data/test_table/key=2", mode="overwrite")

# Read the partitioned table
mergedDF = (
    spark
        .read
        .option("mergeSchema", "true")
        .parquet("data/test_table")
)

Code 
def metadata_to_dict(md, include_row_groups=False):
    d = {
        "created_by": getattr(md, "created_by", None),
        "format_version": getattr(md, "format_version", None),
        "num_columns": md.num_columns,
        "num_row_groups": md.num_row_groups,
        "num_rows": md.num_rows,
        "serialized_size": md.serialized_size,
        "schema": str(md.schema),
        "metadata": {k.decode("utf-8", errors="replace"): v.decode("utf-8", errors="replace") 
                     for k, v in (md.metadata or {}).items()},
    }
    if include_row_groups:
        d["row_groups"] = [
            {"num_rows": md.row_group(i).num_rows} 
            for i in range(md.num_row_groups)
        ]
    return d
